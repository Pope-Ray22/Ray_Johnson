import matplotlib.pyplot as plt
import numpy as np
from sklearn.gaussian_process.kernels import RBF
from sklearn.gaussian_process import GaussianProcessRegressor

T = np.array([700, 900, 1000])
CC = np.array([1e18, 1e19, 4e19]) / 1e17

x_data = T
y_data = CC

sigma_0 = 178e2   #Initial variance guess
sigma_n = 47e2 #Initial noise variance guess
length_scale = 100 #Initial length scale guess

# squared exponential kernel
kernel = 1*RBF(length_scale, length_scale_bounds = (100,1e4))

# normalized y confirms mean is = zero

model = GaussianProcessRegressor(kernel = kernel, alpha = sigma_n, normalize_y = True)

# x values that will be calculated
x_test = np.linspace(500, 1100, 30)

model.fit(x_data[:, np.newaxis], y_data)

y_test, err = model.predict(x_test[:, np.newaxis], return_std = True)

max_fn = np.max(y_test)
print(max_fn)

min_fn = np.min(y_test)
print(min_fn)


plt.scatter(x_data,y_data, label='Data')

plt.scatter(x_test, y_test, label = 'New Data')
plt.fill_between(x_test, y_test+err, y_test-err, alpha = .3)
plt.xlabel('Temperature (C)')
plt.ylabel('Carbon Concentration [C] $cm^3$')
plt.title('Gaussian Process Priors using Squared Exponential Kernel')
plt.legend()
plt.show()

from sklearn.gaussian_process.kernels import RBF
kernels = [sigma_0 * RBF(length_scale=100, length_scale_bounds=(100, 1e4))]
for kernel in kernels:
    # Specify Gaussian Process
    model = GaussianProcessRegressor(kernel, alpha = sigma_n, normalize_y = True)

    x_test = np.linspace(500, 1500, 100)

    model.fit(x_data[:, np.newaxis], y_data)

    y_test, err = model.predict(x_test[:, np.newaxis], return_std = True)
    
    # Plot posterior
    plt.figure(figsize=(8, 8))
    plt.subplot(2, 1, 1)
    plt.plot(x_test, y_test)
    plt.fill_between(x_test, y_test - err, y_test + err, alpha=0.3, color='k')
    y_samples = y_data
    plt.scatter(x_data, y_samples)
    plt.xlim(500, 1500)
    plt.ylim(-200, 500)
    plt.title("Prior (kernel:  %s, noise: %s)" %(kernel, model.alpha), fontsize=12)

    # Generate data and fit GP
    X = x_data
    y = y_data
    model.fit(X[:, np.newaxis], y)

    # Plot posterior
    plt.subplot(2, 1, 2)
    X_ = np.linspace(500, 1500, 100)
    y_mean, y_std = model.predict(X_[:, np.newaxis], return_std=True)
    plt.plot(X_, y_mean, 'k', lw=3, zorder=9)
    plt.fill_between(X_, y_mean - y_std, y_mean + y_std,
                     alpha=0.3, color='k')

    y_samples = model.sample_y(X_[:, np.newaxis], 10)
    plt.plot(X_, y_samples, lw=1)
    plt.scatter(X, y, c='r', s=50, zorder=10, edgecolors=(0, 0, 0))
    plt.xlim(500, 1500)
    plt.ylim(-500, 1200)
    plt.title("Posterior (kernel: %s, noise: %s)\n Log-Likelihood: %.3f"
              % (model.kernel_, model.alpha, model.log_marginal_likelihood(model.kernel_.theta)),
              fontsize=12)
    plt.tight_layout()

plt.show()

def Q(x,y,y_max):
    # Max f_np1 - Max f_n
    
    for i in range (len(x)):
        if y[i] > y_max:
            Q = y[i]-y_max
        else:
            Q = 0
        return Q
        
    
def MCEI_Acquisition_function(x, B_n, y_max, num_mc_samples):
    for t in range(len(x)):
        all_q_values = []
        for i in range(num_mc_samples):
            y_mean, y_std = B_n.predict(x[t, np.newaxis, np.newaxis], return_std=True)
            f_star = np.random.normal(y_mean, y_std)
            W = np.random.normal(0,47**2)
            y = f_star + W
        
            q = Q(x, y, y_max)
        
            all_q_values.append(q)
        
        score[t] = np.mean(all_q_values)
        
    return score
